class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        output=""
        for i in range(len(nums)):
            nums[i] = str(nums[i])
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                if nums[i]+nums[j] < nums[j]+nums[i]:
                    nums[i], nums[j] = nums[j], nums[i]
        for n in nums:
            output += n
        if output[0] == "0":
            output = "0"

        return output

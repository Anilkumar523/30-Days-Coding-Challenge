from typing import List

class Solution:
    def shipWithinDays(self, weights: List[int], days: int) -> int:
        def canShip(capacity: int) -> bool:
            required_days = 1
            current = 0
            for w in weights:
                if current + w > capacity:
                    required_days += 1
                    current = w
                else:
                    current += w
            return required_days <= days
        low, high = max(weights), sum(weights)
        while low < high:
            mid = (low + high) // 2
            if canShip(mid):
                high = mid 
            else:
                low = mid + 1 
        
        return low

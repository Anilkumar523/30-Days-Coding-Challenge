class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        if nums==[]:
            return 0
        else:
            nums.sort()
            res=1
            count=1
            for i in range(1,len(nums)):
                if nums[i]==nums[i-1]+1:
                    count+=1
                    res=max(res,count)
                elif nums[i]!=nums[i-1]:
                    count=1
            return res

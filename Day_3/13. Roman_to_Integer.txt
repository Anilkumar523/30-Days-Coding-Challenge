class Solution:
    def romanToInt(self, s: str) -> int:
        dici = {"I":1, "V":5, "X":10, "L":50, "C":100, "D":500, "M":1000}
        count = 0
        
        for i in range(len(s)):
            if i < len(s) - 1 and dici[s[i]] < dici[s[i+1]]:
                count -= dici[s[i]]
            else:
                count += dici[s[i]]
        
        return count

class Solution:
    def addTwoNumbers(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        head = ListNode(0)    
        tail = head            
        carry_over = 0        
        while list1 or list2 or carry_over:
            digit1 = list1.val if list1 else 0
            digit2 = list2.val if list2 else 0

            total_sum = digit1 + digit2 + carry_over
            carry_over = total_sum // 10
            tail.next = ListNode(total_sum % 10)
            tail = tail.next
            if list1: 
                list1 = list1.next
            if list2: 
                list2 = list2.next

        return head.next
